Laravel JSON endpoint starter

Starter code for creating json endpoint in Laravel

laravel-json-endpoint transforms to 

use Illuminate\Support\Facades\Log;
use Illuminate\Support\Str;
use Throwable;
public function foo()
{
    try {
       return response()->json([
            "status" => true,
            "message" => "Success message.",
            "data" => [
                "foo" => "bar",
                "foo2" => "bar2",
            ],
        ]);
    } catch (Throwable $t) {
        $logCode = strtoupper(date("Ymd") . Str::random(7));
        Log::error($logCode . " " . $t);
        return response()->json(
            [
                "status" => false,
                "message" =>
                    "Error encountered. You can check log code " .
                    $logCode .
                    " for details.",
                "logCode" => $logCode,
            ],
            500
        );
    }
}

Alphanumeric Input Only

Only allow alphanumeric characters on textbox via javascript

alphanumeric-only transforms to 

const element = document.getElementById('some-id');
const regex = /[^A-Za-z0-9]/g;
element.addEventListener('input', (event) => {
	const currentValue = element.value;
	const updatedValue = currentValue.replace(regex, "");
	element.value = updatedValue;
});
